# Define target name and output directory
init_target (EC_3DGizmo)

# Define source files
file (GLOB CPP_FILES *.cpp)
file (GLOB H_FILES *.h)
file (GLOB XML_FILES *.xml)
file (GLOB MOC_FILES EC_3DGizmo.h)

# Qt4 Moc files to subgroup "CMake Moc"
MocFolder ()

set (SOURCE_FILES ${CPP_FILES} ${H_FILES})

# Qt4 Wrap
QT4_WRAP_CPP(MOC_SRCS ${MOC_FILES})

use_package (BOOST)
use_package (OGRE)
use_package (QT4)
use_core_modules (Core Foundation Interfaces SceneManager RexCommon OgreRenderingModule EntityComponents/EC_OpenSimPrim) 

# Note RexCommon and EC_OpenSimPrim are removed! This project wont build without modifications

build_library (${TARGET_NAME} STATIC ${SOURCE_FILES} ${MOC_SRCS} )

link_package (BOOST)
link_package (QT4)
link_ogre()
link_modules (Core Foundation Interfaces SceneManager RexCommon OgreRenderingModule EC_OpenSimPrim)

SetupCompileFlagsWithPCH()

final_target ()
